        function setupKeyboardNavigation() {

                        exitFullscreenMode();
                        return;

                    transition: opacity 0.3s ease;
                    text-align: center;
                    box-shadow: 0 4px 20px rgba(0,0,0,0.3);
                `;
                document.body.appendChild(hintElement);
            }
            
            // и®ҫзҪ®жҸҗзӨәеҶ…е®№
            let icon = '';
            if (direction === 'left') icon = 'в¬…пёҸ ';
            else if (direction === 'right') icon = ' вһЎпёҸ';
            else icon = 'вҡ пёҸ ';
            
            hintElement.innerHTML = `${icon}${text}`;
            hintElement.style.opacity = '1';
            
            // жё…йҷӨд№ӢеүҚзҡ„е®ҡж—¶еҷЁ
            if (window.keyboardHintTimeout) {
                clearTimeout(window.keyboardHintTimeout);
            }
            
            // 1.5 з§’еҗҺйҡҗи—Ҹ
            window.keyboardHintTimeout = setTimeout(() => {
                hintElement.style.opacity = '0';
                setTimeout(() => {
                    if (hintElement && hintElement.parentNode) {
                        hintElement.remove();
                    }
                }, 300);
            }, 1500);
        }

        // еҠ иҪҪйҳ…иҜ»и®ҫзҪ®
        function loadReadingSettings() {
            // д»ҺlocalStorageеҠ иҪҪи®ҫзҪ®
            const savedFontSize = localStorage.getItem('fontSize');
            const savedTheme = localStorage.getItem('darkMode');
            
            if (savedFontSize) {
                currentFontSize = parseInt(savedFontSize);
                updateFontSize();
            }
            
            if (savedTheme !== null) {
                isDarkMode = savedTheme === 'true';
                updateTheme();
            }
            
            // жӣҙж–°и®ҫзҪ®йқўжқҝжҳҫзӨә
            document.getElementById('fontSizeDisplay').textContent = currentFontSize + 'px';
            document.getElementById('themeToggle').innerHTML = 
                isDarkMode ? 'вҳҖпё?дә®иүІжЁЎејҸ' : 'рҹҢҷ жҡ—й»‘жЁЎејҸ';
        }

        // дҝқеӯҳйҳ…иҜ»и®ҫзҪ®
        function saveReadingSettings() {
            localStorage.setItem('fontSize', currentFontSize.toString());
            localStorage.setItem('darkMode', isDarkMode.toString());
        }

        // и°ғж•ҙеӯ—дҪ“еӨ§е°Ҹ
        function adjustFontSize(change) {
            const newSize = currentFontSize + change;
            if (newSize >= 12 && newSize <= 24) {
                currentFontSize = newSize;
                updateFontSize();
                document.getElementById('fontSizeDisplay').textContent = currentFontSize + 'px';
                saveReadingSettings();
            }
        }

        // жҳҫзӨәз« иҠӮеҶ…е®№
        function displayChapter(detailView) {
            const chapter = detailView.chapter;
            
            // вң?дҝқеӯҳе°ҸиҜҙе’Ңз« иҠӮж ҮйўҳеҲ°е…ЁеұҖеҸҳйҮҸпјҲз”ЁдәҺеҺҶеҸІи®°еҪ•дҝқеӯҳпјү

            console.log('рҹ“– еҪ“еүҚз« иҠӮж Үйўҳ:', currentChapterTitle);

            document.getElementById('novelLink').textContent = detailView.novelTitle;
            document.getElementById('currentChapter').textContent = chapter.title;
            
            // жҳҫзӨәз« иҠӮеәҸеҸ·ж ҮиҜҶ
            const chapterNumberBadge = document.getElementById('chapterNumberBadge');
            chapterNumberBadge.textContent = `з¬?{chapter.chapterNumber}з« `;
            chapterNumberBadge.style.display = 'inline-block';
            
            // жҳҫзӨәз« иҠӮдҝЎжҒҜ
            document.getElementById('chapterTitle').textContent = chapter.title;
            document.getElementById('chapterInfo').textContent = 
                `з¬?{chapter.chapterNumber}з«?| ` +
                `еҲӣе»әж—¶й—ҙ: ${new Date(chapter.createTime).toLocaleString()}`;
            
            // еӨ„зҗҶз« иҠӮжҸ’еӣҫ
            handleChapterImage(chapter);
            
            // жёІжҹ“ Markdown еҶ…е®№
            const content = chapter.content || 'жҡӮж— еҶ…е®№';
            // дҝқеӯҳеҺҹе§ӢеҶ…е®№з”ЁдәҺиҮӘеҠЁй…Қеӣҫ
            currentChapterContent = content;
            try {
                if (typeof marked !== 'undefined') {
                    // marked 12.x зүҲжң¬зӣҙжҺҘдҪҝз”Ё marked() еҮҪж•°
                    document.getElementById('chapterText').innerHTML = marked.parse(content);
                } else {
                    // йҷҚзә§еӨ„зҗҶпјҡзӣҙжҺҘжҳҫзӨәж–Үжң¬еҶ…е®?                    console.warn('Marked.js дёҚеҸҜз”ЁпјҢзӣҙжҺҘжҳҫзӨәж–Үжң¬еҶ…е®№');

                }

                // еҮәй”ҷж—¶зӣҙжҺҘжҳҫзӨәеҺҹе§ӢеҶ…е®?                document.getElementById('chapterText').textContent = content;
            }
            
            // й«ҳдә®д»Јз Ғеқ?            document.querySelectorAll('pre code').forEach((block) => {
                hljs.highlightElement(block);
            });
            
            // и®ҫзҪ®еҜјиҲӘжҢүй’®зҠ¶жҖ?            const prevButton = document.getElementById('prevButton');
            const nextButton = document.getElementById('nextButton');
            
            if (detailView.hasPrevious && detailView.previousChapterId) {
                prevButton.disabled = false;
                prevButton.onclick = () => goToChapter(detailView.previousChapterId);

                prevButton.disabled = true;
                window.goToPrevious = null; // зҰҒз”Ёж—¶и®ҫзҪ®дёә null
            }
            
            if (detailView.hasNext && detailView.nextChapterId) {
                nextButton.disabled = false;
                nextButton.onclick = () => goToChapter(detailView.nextChapterId);

            } else {

                window.goToNext = null; // зҰҒз”Ёж—¶и®ҫзҪ®дёә null
            }
            
            // йҡҗи—ҸеҠ иҪҪзҠ¶жҖҒпјҢжҳҫзӨәеҶ…е®№

        // и®ҫзҪ®зӘ—еҸЈеӨ§е°ҸеҸҳеҢ–зӣ‘еҗ¬
        function setupResizeObserver() {
            // дҪҝз”Ё ResizeObserver зӣ‘еҗ¬зӘ—еҸЈеӨ§е°ҸеҸҳеҢ–
            let resizeTimeout;
            
            window.addEventListener('resize', function() {
                // йҳІжҠ–еӨ„зҗҶпјҢйҒҝе…Қйў‘з№Ғи§ҰеҸ?                clearTimeout(resizeTimeout);
                resizeTimeout = setTimeout(() => {
                    if (isCardMode) {
                        adjustCardToWindow();
                    }
                }, 250);
            });
        }
        
        // ж №жҚ®зӘ—еҸЈеӨ§е°Ҹи°ғж•ҙеҚЎзүҮе°әеҜё
        function adjustCardToWindow() {
            const cardDisplayArea = document.querySelector('.card-display-area');
            const contentCard = document.getElementById('contentCard');
            
            if (!cardDisplayArea || !contentCard) return;
            
            const windowHeight = window.innerHeight;
            const windowWidth = window.innerWidth;
            
            // и®Ўз®—еҗҲйҖӮзҡ„еҚЎзүҮй«ҳеәҰпјҲи§ҶеҸЈзҡ„ 50%-75%пј?            const idealHeight = Math.floor(windowHeight * 0.65);
            
            // и®ҫзҪ®жңҖе°Ҹе’ҢжңҖеӨ§й«ҳеәҰйҷҗеҲ?            const minHeight = Math.max(300, windowHeight * 0.4);
            const maxHeight = Math.min(idealHeight, windowHeight * 0.75 - 100);
            
            contentCard.style.minHeight = minHeight + 'px';
            contentCard.style.maxHeight = maxHeight + 'px';
            
            // ж №жҚ®зӘ—еҸЈе®ҪеәҰи°ғж•ҙеҶ…иҫ№и·?            if (windowWidth < 480) {
                contentCard.style.padding = '20px 15px';
            } else if (windowWidth < 768) {
                contentCard.style.padding = '30px 25px';
            } else {
                contentCard.style.padding = '40px';
            }
            
            console.log(`зӘ—еҸЈи°ғж•ҙпјҡе®Ҫ${windowWidth}px, й«?{windowHeight}px, еҚЎзүҮй«ҳеәҰпј?{minHeight}-${maxHeight}px`);
        }

        // жҷәиғҪеҲҶеүІеҶ…е®№дёәеӨҡдёӘеҚЎзү?        function splitContentIntoCards(htmlContent) {
            const cards = [];
            
            // еҲӣе»әдёҙж—¶ div жқҘи§Јжһ?HTML
            const tempDiv = document.createElement('div');
            tempDiv.innerHTML = htmlContent;
            
            // иҺ·еҸ–жүҖжңүж®өиҗҪе’Ңеқ—зә§е…ғзҙ 
            const paragraphs = Array.from(tempDiv.children);
            
            if (paragraphs.length === 0) {
                // еҰӮжһңжІЎжңүж®өиҗҪпјҢе°қиҜ•жҢүеҸҘеҸ·еҲҶеүІ
                const text = tempDiv.textContent || htmlContent;
                return splitBySentence(text);
            }
            
            // ж №жҚ®зӘ—еҸЈеӨ§е°ҸеҠЁжҖҒи°ғж•ҙжҜҸеҚЎзүҮзҡ„еӯ—з¬Ұж•°пјҲеҮҸе°‘иЎҢж•°пјү
            const windowHeight = window.innerHeight;
            let targetLengthPerCard = 400; // й»ҳи®ӨеҖјпјҲд»?600 еҮҸе°‘еҲ?400пј?            
            // ж №жҚ®и§ҶеҸЈй«ҳеәҰи°ғж•ҙзӣ®ж Үй•ҝеәҰ
            if (windowHeight < 600) {
                targetLengthPerCard = 300; // е°ҸеұҸе№•иҝӣдёҖжӯҘеҮҸе°‘еӯ—ж•?            } else if (windowHeight > 900) {
                targetLengthPerCard = 500; // еӨ§еұҸе№•зЁҚеҫ®еўһеҠ пјҢдҪҶд№ҹдёҚи¶…иҝ?500
            }
            
            let currentCard = '';
            let currentLength = 0;
            
            for (let i = 0; i < paragraphs.length; i++) {
                const paragraph = paragraphs[i];
                const paragraphHtml = paragraph.outerHTML;
                const paragraphText = paragraph.textContent || '';
                const paragraphLength = paragraphText.length;
                
                // еҰӮжһңеҚ•дёӘж®өиҗҪе°ұи¶…иҝҮзӣ®ж Үй•ҝеәҰпјҢйңҖиҰҒиҝӣдёҖжӯҘеҲҶеү?                if (paragraphLength > targetLengthPerCard * 1.2) {  // йҷҚдҪҺйҳҲеҖјеҲ° 1.2 еҖ?                    // еҰӮжһңеҪ“еүҚеҚЎзүҮе·ІжңүеҶ…е®№пјҢе…Ҳдҝқеӯҳ
                    if (currentCard.trim()) {
                        cards.push(currentCard);
                        currentCard = '';
                        currentLength = 0;
                    }
                    
                    // еҲҶеүІй•ҝж®өиҗ?                    const subCards = splitLongParagraph(paragraphHtml, paragraphText, targetLengthPerCard);
                    cards.push(...subCards);
                } else {
                    // жЈҖжҹҘж·»еҠ еҲ°еҪ“еүҚеҚЎзүҮжҳҜеҗҰдјҡи¶…еҮәпјҲйҷҚдҪҺйҳҲеҖјпјү
                    if (currentLength + paragraphLength > targetLengthPerCard && currentCard.trim()) {
                        // дҝқеӯҳеҪ“еүҚеҚЎзүҮ
                        cards.push(currentCard);
                        currentCard = paragraphHtml;
                        currentLength = paragraphLength;
                    } else {
                        // ж·»еҠ еҲ°еҪ“еүҚеҚЎзү?                        currentCard += paragraphHtml;
                        currentLength += paragraphLength;
                    }
                }
            }
            
            // ж·»еҠ жңҖеҗҺдёҖдёӘеҚЎзү?            if (currentCard.trim()) {
                cards.push(currentCard);
            }
            
            console.log(`еҶ…е®№еҲҶеүІе®ҢжҲҗпјҡе…ұ${cards.length}еј еҚЎзүҮпјҢжҜҸеҚЎзүҮзәҰ${targetLengthPerCard}еӯ—`);
            return cards;
        }
        
        // жҢүеҸҘеӯҗеҲҶеүІй•ҝж®өиҗҪ
        function splitLongParagraph(paragraphHtml, paragraphText, maxLength) {
            const result = [];
            
            // е°қиҜ•жҢүдёӯж–ҮеҸҘеҸ·гҖҒй—®еҸ·гҖҒж„ҹеҸ№еҸ·еҲҶеүІ
            const sentences = paragraphText.split(/([.!?.])/g);
            
            let currentPart = '';
            let currentLength = 0;
            
            for (let i = 0; i < sentences.length; i += 2) {
                const sentence = sentences[i] + (sentences[i+1] || '');
                const sentenceLength = sentence.length;
                
                if (currentLength + sentenceLength > maxLength && currentPart.trim()) {
                    // еҢ…иЈ…дёәж®өиҗ?HTML
                    result.push('<p>' + currentPart.trim() + '</p>');
                    currentPart = sentence;
                    currentLength = sentenceLength;
                } else {
                    currentPart += sentence;
                    currentLength += sentenceLength;
                }
            }
            
            if (currentPart.trim()) {
                result.push('<p>' + currentPart.trim() + '</p>');
            }
            
            return result.length > 0 ? result : ['<p>' + paragraphText.substring(0, maxLength) + '...</p>'];
        }
        
        // жҢүеҸҘеӯҗеҲҶеүІж–Үжң?        function splitBySentence(text) {
            const sentences = text.split(/([.!?.\n])/g).filter(s => s.trim());
            const cards = [];
            let currentPart = '';
            const maxLength = 500;
            
            for (let i = 0; i < sentences.length; i += 2) {
                const sentence = sentences[i] + (sentences[i+1] || '');
                if ((currentPart + sentence).length > maxLength && currentPart.trim()) {
                    cards.push(currentPart.trim());
                    currentPart = sentence;
                } else {
                    currentPart += sentence;
                }
            }
            
            if (currentPart.trim()) {
                cards.push(currentPart.trim());
            }
            
            return cards.length > 0 ? cards : [text.substring(0, 1000)];
        }

        // д»ҺйҖүдёӯж–Үжң¬дёӯиҜҶеҲ«дәәеҗҚе№¶иҺ·еҸ–еӨ–иІҢжҸҸеҶҷ
        function extractCharacterAppearances(selectedText) {
            const appearances = [];
            
            currentChapterCharacterCards.forEach(card => {
                const characterName = card.name;
                if (characterName && selectedText.includes(characterName)) {
                    // жҸҗеҸ–еӨ–иІҢжҸҸеҶҷ
                    let appearanceDescription = '';
                    
                    if (card.appearance) {
                        const app = card.appearance;
                        const parts = [];
                        parts.push(`е№ҙйҫ„пј?{card.age}`);
                        if (app.height) parts.push(`иә«й«ҳ${app.height}`);
                        if (app.hair) parts.push(`еҸ‘еһӢпј?{app.hair}`);
                        if (app.eyes) parts.push(`зңјзқӣпј?{app.eyes}`);
                        if (app.build) parts.push(`дҪ“еһӢпј?{app.build}`);
                        if (app.distinguishingFeatures) parts.push(`зү№еҫҒпј?{app.distinguishingFeatures}`);
                        appearanceDescription = parts.join('пј?);
                    }
                    
                    if (card.appearanceSummary) {
                        appearanceDescription += 'гҖ? + card.appearanceSummary;
                    }
                    
                    if (appearanceDescription) {
                        appearances.push({
                            name: characterName,
                            description: appearanceDescription.trim()
                        });
                        console.log(`рҹ‘Ө иҜҶеҲ«еҲ°и§’иү?[${characterName}]:`, appearanceDescription);
                    }
                }
            });
            
            return appearances;
        }

        function buildAIPrompt(inputText) {
            const characterAppearances = extractCharacterAppearances(inputText);

            // иҺ·еҸ–йҖүдёӯж–Үеӯ—зҡ„еүҚеҗҺж–ҮдёҠдёӢж–?            let contextBefore = '';
            let contextAfter = '';
            if (currentChapterContent && currentChapterContent.includes(inputText)) {
                const selectedIndex = currentChapterContent.indexOf(inputText);
                const contextLength = 800; // еүҚеҗҺеҗ„еҸ–800еӯ?
                // жҸҗеҸ–еүҚж–Ү
                const beforeStart = Math.max(0, selectedIndex - contextLength);
                contextBefore = currentChapterContent.substring(beforeStart, selectedIndex);

                // жҸҗеҸ–еҗҺж–Ү
                const afterEnd = Math.min(currentChapterContent.length, selectedIndex + inputText.length + contextLength);
                contextAfter = currentChapterContent.substring(selectedIndex + inputText.length, afterEnd);
            }

            // дҪҝз”Ёж–°зҡ„з»“жһ„еҢ–ж јеј?- е…ҲеҲҶжһҗдёҠдёӢж–ҮпјҢеҶҚз”ҹжҲҗжҸҗзӨәиҜ?            let promptBuilder = `# и§’иүІпјҡе°ҸиҜҙеңәжҷҜи§Ҷи§үеҲҶжһҗдё“е®¶\n\n`;

            promptBuilder += `## д»»еҠЎиҜҙжҳҺ\n`;
            promptBuilder += `дҪ йңҖиҰҒеҲҶдёӨжӯҘе®ҢжҲҗд»»еҠЎпјҡ\n`;
            promptBuilder += `з¬¬дёҖжӯҘпјҡж №жҚ®жҸҗдҫӣзҡ„е°ҸиҜҙдёҠдёӢж–ҮпјҢиҜҰз»ҶеҲҶжһҗйҖүдёӯеңәжҷҜдёӯзҡ„дәәзү©дҝЎжҒҜгҖҒзү©е“ҒйҒ“е…·гҖҒзҺҜеўғиғҢжҷҜзӯүпјӣ\n`;
            promptBuilder += `з¬¬дәҢжӯҘпјҡеҹәдәҺеҲҶжһҗз»“жһңпјҢз”ҹжҲҗдёҖдёӘдё“дёҡзҡ„AIж–Үз”ҹеӣҫжҸҗзӨәиҜҚпјҲиӢұж–ҮпјүгҖӮ\n\n`;

            promptBuilder += `## е°ҸиҜҙе…Ёж–ҮдёҠдёӢж–Ү\n`;
            if (contextBefore) {
                promptBuilder += `гҖҗеүҚж–ҮгҖ?..${contextBefore}...\n\n`;
            }
            promptBuilder += `гҖҗйҖүдёӯзҡ„ж ёеҝғеңәжҷҜгҖ?{inputText}\n\n`;
            if (contextAfter) {
                promptBuilder += `гҖҗеҗҺж–ҮгҖ?..${contextAfter}...\n\n`;
            }

            promptBuilder += `## з¬¬дёҖжӯҘпјҡеңәжҷҜж·ұеәҰеҲҶжһҗ\n`;
            promptBuilder += `иҜ·еҹәдәҺд»ҘдёҠе°ҸиҜҙеҶ…е®№пјҢиҜҰз»ҶеҲҶжһҗпјҡ\n\n`;
            promptBuilder += `1. **еҮәеңәдәәзү©еҲҶжһҗ**пјҡ\n`;
            promptBuilder += `   - дәәзү©е§“еҗҚ\n`;
            promptBuilder += `   - еӨ–иІҢзү№еҫҒпјҲи„ёеһӢгҖҒдә”е®ҳгҖҒеҸ‘иүІгҖҒзһіиүІгҖҒиә«й«ҳдҪ“еһӢзӯүпјү\n`;

            promptBuilder += `   - еҠЁдҪңе§ҝжҖҒпјҲжӯЈеңЁеҒҡд»Җд№ҲеҠЁдҪңгҖҒиЎЁжғ…еҰӮдҪ•пјү\n\n`;

        // ејҖе§ӢиҪ®иҜўд»»еҠЎзҠ¶жҖ?        function startTaskPolling(taskId) {
            autoImagePollInterval = setInterval(async () => {
                try {
                    const response = await fetch(`/api/chapter-image/auto-generate/status/${taskId}`);
                    const result = await response.json();

                    if (result.success && result.task) {
                        updateTaskProgress(result.task);

                        // жЈҖжҹҘд»»еҠЎжҳҜеҗҰе®ҢжҲ?                        if (result.task.status === 'COMPLETED' || result.task.status === 'FAILED') {
                            clearInterval(autoImagePollInterval);
                            autoImagePollInterval = null;

                            if (result.task.status === 'COMPLETED') {
                                // ж”¶йӣҶз”ҹжҲҗзҡ„еӣҫзүҮж•°жҚ?                                generatedImagesData = result.task.items
                                    .filter(item => item.status === 'COMPLETED' && item.imageUrl)
                                    .map(item => ({
                                        imageUrl: item.imageUrl,
                                        prompt: item.prompt,
                                        position: item.position,
                                        insertIndex: currentImagePositions[item.index]?.insertIndex || 0
                                    }));

                                // е»¶иҝҹдёҖдёӢеҶҚжҳҫзӨәз»“жһңпјҢи®©з”ЁжҲ·зңӢеҲ°е®ҢжҲҗзҠ¶жҖ?                                setTimeout(() => {
                                    document.getElementById('generatingLoading').style.display = 'none';
                                    document.getElementById('imageResult').style.display = 'block';
                                    displayGeneratedImages();
                                }, 1000);
                            } else {
                                alert('з”ҹжҲҗеӨұиҙҘпј? + (result.task.errorMessage || 'жңӘзҹҘй”ҷиҜҜ'));
                                document.getElementById('generatingLoading').style.display = 'none';
                            }
                        }
                    }
                } catch (error) {
                    console.error('иҪ®иҜўд»»еҠЎзҠ¶жҖҒеӨұиҙ?', error);
                }
            }, 1500); // жҜ?.5з§’жҹҘиҜўдёҖж¬?        }

        // жӣҙж–°д»»еҠЎиҝӣеәҰжҳҫзӨә
        function updateTaskProgress(task) {
            const total = task.totalCount || 0;
            const completed = task.completedCount || 0;
            const percentage = total > 0 ? Math.round((completed / total) * 100) : 0;

            // жӣҙж–°жҖ»дҪ“иҝӣеәҰ
            document.getElementById('generatingProgressBar').style.width = percentage + '%';
            document.getElementById('generatingProgressText').textContent = `иҝӣеәҰпј?{completed}/${total}`;
            document.getElementById('generatingText').textContent =
                task.status === 'PROCESSING' ? 'жӯЈеңЁз”ҹжҲҗеӣҫзүҮ...' :
                task.status === 'COMPLETED' ? 'з”ҹжҲҗе®ҢжҲҗпј? :
                task.status === 'FAILED' ? 'з”ҹжҲҗеӨұиҙҘ' : 'еҮҶеӨҮдё?..';

            // жӣҙж–°жҜҸдёӘйЎ№зӣ®зҡ„иҝӣеә?            if (task.items) {
                task.items.forEach((item, idx) => {
                    updatePromptItem(idx, item);
                    updateImageItem(idx, item);
                });
            }
        }

        // жӣҙж–°жҸҗзӨәиҜҚйЎ№зӣ®жҳҫзӨ?        function updatePromptItem(idx, item) {
            const promptItem = document.getElementById(`prompt-item-${idx}`);
            const promptStatus = document.getElementById(`prompt-status-${idx}`);
            const promptText = document.getElementById(`prompt-text-${idx}`);

            if (!promptItem) return;

            const statusClass = item.status?.toLowerCase() || 'pending';
            promptItem.className = `progress-item ${statusClass}`;

            const statusMap = {
                'PENDING': 'зӯүеҫ…дё?,
                'GENERATING': 'з”ҹжҲҗдё?,
                'COMPLETED': 'е·Іе®ҢжҲ?,
                'FAILED': 'еӨұиҙҘ'
            };
            promptStatus.textContent = statusMap[item.status] || 'зӯүеҫ…дё?;
            promptStatus.className = `progress-item-status ${statusClass}`;

            if (item.prompt) {
                promptText.textContent = item.prompt.substring(0, 150) + '...';
            }
        }

                container.appendChild(item);
            });
        }
        
        // дҝқеӯҳеӣҫзүҮеҲ°з« иҠӮеҶ…е®?        async function saveImagesToContent() {
            try {
                if (!currentChapterContent) {
                    alert('з« иҠӮеҶ…е®№дёәз©ә');
                    return;
                }
                
                // и°ғз”ЁеҗҺз«ҜжҺҘеҸЈжҸ’е…ҘеӣҫзүҮ
                const response = await fetch('/api/chapter-image/insert', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        content: currentChapterContent,
                        images: generatedImagesData
                    })
                });
                
                const result = await response.json();
                
                if (result.success) {
                    // дҝқеӯҳжӣҙж–°еҗҺзҡ„еҶ…е®№еҲ°ж•°жҚ®еә“
                    await saveUpdatedContent(result.newContent);
                    
                    // е…ій—ӯйқўжқҝ
                    closeAutoImagePanel();
                    
                    alert('вң?еӣҫзүҮе·ІжҲҗеҠҹжҸ’е…ҘеҲ°з« иҠӮеҶ…е®№дёӯпјҒ');
                } else {
                    throw new Error(result.error || 'дҝқеӯҳеӨұиҙҘ');
                }
                
            } catch (error) {
                console.error('дҝқеӯҳеӣҫзүҮеӨұиҙҘ:', error);
                alert('дҝқеӯҳеӨұиҙҘпј? + error.message);
            }
        }
